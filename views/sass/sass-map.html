# Sass map

# 1. map.deep-merge($map1, $map2)

  • 语法：
map.deep-merge($map1, $map2)  => map
  • 使用
$helvetica-light: (
  "weights": (
    "lightest": 100,
    "light": 300
  )
);
$helvetica-heavy: (
  "weights": (
    "medium": 500,
    "bold": 700
  )
);

@debug map.deep-merge($helvetica-light, $helvetica-heavy);
// (
//   "weights": (
//     "lightest": 100,
//     "light": 300,
//     "medium": 500,
//     "bold": 700
//   )
// )

# 2. map.deep-remove

  • 语法：
map.deep-remove($map, $key, $keys...) //=> map
  • 1. 使用：如果$keys为空，则返回的副本，其中$map没有与关联的值 $key。
$font-weights: ("regular": 400, "medium": 500, "bold": 700);

@debug map.deep-remove($font-weights, "regular");
// ("medium": 500, "bold": 700)
  • 2. 使用：
    • 如果$keys不为空，请从左到右跟随键集（包括$key和排除最后一个键）$keys从左到右查找要更新的嵌套地图。返回$map目标地图没有与中的最后一个键关联的值 的副本$keys。 **
$fonts: (
  "Helvetica": (
    "weights": (
      "regular": 400,
      "medium": 500,
      "bold": 700
    )
  )
);

@debug map.deep-remove($fonts, "Helvetica", "weights", "regular");
// (
//   "Helvetica": (
//     "weights: (
//       "medium": 500,
//       "bold": 700
//     )
//   )
// )

# 3. map.get 和 map-get

  • 语法：
map.get($map, $key, $keys...)
map-get($map, $key, $keys...)

  • 1. 使用:

    • 如果$keys为空，则返回与中$map关联 的值$key。

    • 如果$map没有与关联的值$key，则返回null (opens new window)。

$font-weights: ("regular": 400, "medium": 500, "bold": 700);

@debug map.get($font-weights, "medium"); // 500
@debug map.get($font-weights, "extra-bold"); // null
  • 2. 使用
  • 如果$keys不为空，请从左到右跟随键集（包括$key和排除最后一个键）$keys从左到右查找要搜索的嵌套地图。返回目标地图中与中的最后一个键关联的值 $keys。
$fonts: (
  "Helvetica": (
    "weights": (
      "regular": 400,
      "medium": 500,
      "bold": 700
    )
  )
);

@debug map.get($fonts, "Helvetica", "weights", "regular"); // 400
@debug map.get($fonts, "Helvetica", "colors"); // null

# 4. map.has-key 和 map-has-key

  • 语法：
map.has-key($map, $key, $keys...)
map-has-key($map, $key, $keys...) //=> boolean
  • 如果$keys为空，则返回是否$map包含与关联的值 $key。
$font-weights: ("regular": 400, "medium": 500, "bold": 700);

@debug map.has-key($font-weights, "regular"); // true
@debug map.has-key($font-weights, "bolder"); // false
  • 2. 使用
    • 如果$keys不为空，请从左到右跟随键集（包括$key和排除最后一个键）$keys从左到右查找要搜索的嵌套地图。如果目标映射包含与中的最后一个键关联的值，则返回true $keys。
$fonts: (
  "Helvetica": (
    "weights": (
      "regular": 400,
      "medium": 500,
      "bold": 700
    )
  )
);

@debug map.has-key($fonts, "Helvetica", "weights", "regular"); // true
@debug map.has-key($fonts, "Helvetica", "colors"); // false

# 5. map.keys 和 map-keys

  • 语法：
map.keys($map)
map-keys($map) //=> list
  • 使用：返回中所有键的逗号分隔列表 $map。
$font-weights: ("regular": 400, "medium": 500, "bold": 700);

@debug map.keys($font-weights); // "regular", "medium", "bold"

# 6. map.merge 、map-merge

  • 语法
map.merge($map1, $map2)
map-merge($map1, $map2)
map.merge($map1, $keys..., $map2)
map-merge($map1, $keys..., $map2) //=> map
  • 如果没有$keys通过，返回与来自所有键和值的新地图$map1和 $map2。
  • 如果同时$map1与$map2具有相同的钥匙，$map2“价值优先。
  • 返回的映射中所有也以出现的键$map1具有与中相同的顺序$map1。来自的新键$map2显示在地图的末尾。
$light-weights: ("lightest": 100, "light": 300);
$heavy-weights: ("medium": 500, "bold": 700);

@debug map.merge($light-weights, $heavy-weights);
// ("lightest": 100, "light": 300, "medium": 500, "bold": 700)

  • 如果$keys不为空，则遵循$keys来查找目标合并的嵌套地图。如果$keys映射中缺少任何键入或引用的值不是映射，则将该键的值设置为空映射。

  • 返回$map1目标地图被新地图替换的副本，该地图包含目标地图和中的所有键和值 $map2。

$fonts: (
  "Helvetica": (
    "weights": (
      "lightest": 100,
      "light": 300
    )
  )
);
$heavy-weights: ("medium": 500, "bold": 700);

@debug map.merge($fonts, "Helvetica", "weights", $heavy-weights);
// (
//   "Helvetica": (
//     "weights": (
//       "lightest": 100,
//       "light": 300,
//       "medium": 500,
//       "bold": 700
//     )
//   )
// )

# 7. map.remove 和 map-remove

  • 语法
map.remove($map, $keys...)
map-remove($map, $keys...) //=> map

  • 传回的副本，$map而没有与关联的任何值 $keys。

  • 如果key in$keys中没有关联的值$map，则会将其忽略。

$font-weights: ("regular": 400, "medium": 500, "bold": 700);

@debug map.remove($font-weights, "regular"); // ("medium": 500, "bold": 700)
@debug map.remove($font-weights, "regular", "bold"); // ("medium": 500)
@debug map.remove($font-weights, "bolder");
// ("regular": 400, "medium": 500, "bold": 700)

# 8. map.set

  • 语法
map.set($map, $key, $value)
map.set($map, $keys..., $key, $value) //=> map
  • 如果$keys未传递，则返回的副本，$map其值$key设置为 $value。
$font-weights: ("regular": 400, "medium": 500, "bold": 700);

@debug map.set($font-weights, "regular", 300);
// ("regular": 300, "medium": 500, "bold": 700)
  • 如果$keys已通过，则按照$keys来查找要更新的嵌套地图。如果$keys映射中缺少任何键入或引用的值不是映射，则将该键的值设置为空映射。
  • 返回的副本，$map其中目标地图的值$key设置为 $value。
$fonts: (
  "Helvetica": (
    "weights": (
      "regular": 400,
      "medium": 500,
      "bold": 700
    )
  )
);

@debug map.set($fonts, "Helvetica", "weights", "regular", 300);
// (
//   "Helvetica": (
//     "weights": (
//       "regular": 300,
//       "medium": 500,
//       "bold": 700
//     )
//   )
// )

# 9. map.values 和 map-values

  • 语法
map.values($map)
map-values($map) //=> list
  • 返回逗号分隔的列表，其中包含所有值 $map。
$font-weights: ("regular": 400, "medium": 500, "bold": 700);

@debug map.values($font-weights); // 400, 500, 700
